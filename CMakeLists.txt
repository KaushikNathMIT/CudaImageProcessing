cmake_minimum_required(VERSION 3.6)
project(cuda2)
find_package(CUDA REQUIRED)
find_package(OpenCV REQUIRED)
set(CMAKE_CXX_STANDARD 11)

set(SOURCE_FILES test.cu BasicFilter.cpp FaceDetectionCuda.cpp)
cuda_add_executable(cuda2 test.cu)
add_executable(cudaocv BasicFilter.cpp)
add_executable(cudaFaceDetect FaceDetectionCuda.cpp)
add_executable(simpleFaceDetect FaceDetectionSimple.cpp)
include_directories( ${OpenCV_INCLUDE_DIRS} )
target_link_libraries( cudaocv ${OpenCV_LIBS} )
target_link_libraries( cudaFaceDetect ${OpenCV_LIBS} )
target_link_libraries( simpleFaceDetect ${OpenCV_LIBS} )
